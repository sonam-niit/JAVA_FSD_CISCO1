
>sudo docker network create microservice-mysql-network
>sudo docker network ls
Run Mysql image in another container
>sudo docker run --name mysql_container --network microservice-mysql-network -e MYSQL_ROOT_PASSWORD=123456 -d -p 3309:3306  mysql

>sudo docker exec -it mysql_container mysql -uroot -p

Update Application Properties

spring.datasource.url=jdbc:mysql://mysql-container:3306/phaseendproject?createDatabaseIfNotExist=true
spring.datasource.username=root
spring.datasource.password=123456
spring.jpa.hibernate.ddl-auto=update
spring.jpa.properties.hibernate.dialect=org.hibernate.dialect.MySQLDialect
spring.jpa.show-sql=true
server.port=9090

sudo docker run --name pets-container --network microservice-mysql-network -p 9090:9090 -d petsclinic
